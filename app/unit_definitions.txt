# app/unit_definitions.txt

# This file defines custom unit conversions and countable units for the Pint library.
# The app knows standard units (gram, ounce, cup, etc.) by default. This file adds to that knowledge.
# Format: unit_name = value * base_unit

# -----------------------------------------------------------------------------
# Section 1: Standard Volume & Weight Equivalencies
# For common kitchen items with a relatively standard size and weight.
# -----------------------------------------------------------------------------
stick_of_butter = 8 * tablespoon = 0.5 * cup = 4 * ounce = 113 * gram
stick = stick_of_butter  # Alias for convenience

# -----------------------------------------------------------------------------
# Section 2: Approximate 'Folk' or Imprecise Measures
# These have generally accepted (though not perfectly standard) conversions.
# -----------------------------------------------------------------------------
pinch = 1/16 * teaspoon
dash = 1/8 * teaspoon
smidgen = 1/32 * teaspoon
drop = 1/96 * teaspoon  # Approximately 20 drops per ml

# -----------------------------------------------------------------------------
# Section 3: Countable, Dimensionless Units
# These are for items that are counted, not measured by weight/volume.
# Defining them here tells the system they are valid units to count and subtract.
# -----------------------------------------------------------------------------
slice = 1 * dimensionless
slices = slice
each = 1 * dimensionless
clove = 1 * dimensionless
cloves = clove
head = 1 * dimensionless
heads = head
sprig = 1 * dimensionless
sprigs = sprig
bunch = 1 * dimensionless
bunches = bunch
stalk = 1 * dimensionless
stalks = stalk
ear = 1 * dimensionless
ears = ear
fillet = 1 * dimensionless
fillets = fillet
leaf = 1 * dimensionless
leaves = leaf
piece = 1 * dimensionless
pieces = piece
pat = 1 * dimensionless  # e.g., a pat of butter
pats = pat
link = 1 * dimensionless # e.g., sausage links
links = link
strip = 1 * dimensionless # e.g., bacon strips
strips = strip
sheet = 1 * dimensionless # e.g., phyllo or gelatin sheets
sheets = sheet
